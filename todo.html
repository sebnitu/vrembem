Current scripts

The NPM scripts that I've added for each package so far.

Root

// component package.json
"scripts": {
  "build": "lerna run build --parallel",
  "clean": "lerna run clean --parallel",
  "fix": "npm-run-all fix:styles fix:scripts",
  "fix:scripts": "eslint --fix packages",
  "fix:styles": "stylelint --fix \"packages/**/*.scss\"",
  "lint": "npm-run-all lint:styles lint:scripts",
  "lint:scripts": "eslint packages",
  "lint:styles": "stylelint \"packages/**/*.scss\"",
  "watch": "lerna run watch --parallel"
}

Scripts

// component package.json
"scripts": {
  "build": "npm-run-all clean scripts",
  "clean": "del dist",
  "scripts": "npm-run-all scripts:compile scripts:min",
  "scripts:compile": "mkdirp dist && browserify index.js --debug dist/scripts.js -t [ babelify --presets [ @babel/preset-env ] ] | exorcist dist/scripts.js.map > dist/scripts.js",
  "scripts:min": "mkdirp dist && browserify index.js --debug dist/scripts.min.js -t [ babelify --presets [ @babel/preset-env ] ] -p tinyify | exorcist dist/scripts.min.js.map > dist/scripts.min.js",
  "watch": "npm run watch:scripts",
  "watch:scripts": "nodemon -e js -x 'npm run scripts' -i dist"
}

Styles

// component package.json
"scripts": {
  "build": "npm-run-all clean styles",
  "clean": "del dist",
  "styles": "npm-run-all styles:compile styles:prefix styles:min",
  "styles:compile": "sass index.scss dist/styles.css --load-path node_modules",
  "styles:min": "sass dist/styles.css dist/styles.min.css --style compressed",
  "styles:prefix": "postcss dist/styles.css --map --use autoprefixer --output dist/styles.css",
  "watch": "npm run watch:styles",
  "watch:styles": "nodemon -e scss -x 'npm run styles'"
}
